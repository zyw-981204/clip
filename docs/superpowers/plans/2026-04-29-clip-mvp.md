# clip MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a macOS menu-bar clipboard history panel with global hotkey, substring search, pinning, marker-based privacy filtering, and 500/30d retention.

**Architecture:** Single LSUIElement Swift app. Background `PasteboardObserver` polls `NSPasteboard.changeCount`, filters via `PrivacyFilter`, persists to SQLite via `HistoryStore`. Global hotkey opens a `.nonactivatingPanel` SwiftUI list; selection writes to pasteboard + `CGEvent.post(⌘V)` into the still-frontmost target app.

**Tech Stack:** Swift 6.0, macOS 13+, GRDB.swift (SQLite), KeyboardShortcuts (Sindre Sorhus), SwiftUI, AppKit (`NSPanel`, `NSStatusItem`), CoreGraphics (`CGEvent`).

**Spec:** `docs/superpowers/specs/2026-04-29-clip-design.md`

```swift
import Foundation
import CryptoKit

struct ClipItem: Identifiable, Equatable {
    var id: Int64?
    var content: String
    var contentHash: String
    var sourceBundleID: String?
    var sourceAppName: String?
    var createdAt: Int64
    var pinned: Bool
    var byteSize: Int
    var truncated: Bool

    static func byteSize(of s: String) -> Int { s.utf8.count }

    static func truncateIfNeeded(_ s: String, limit: Int) -> (String, Bool) {
        let bytes = Array(s.utf8)
        guard bytes.count > limit else { return (s, false) }
        // Truncate at a UTF-8 codepoint boundary <= limit.
        var cut = limit
        while cut > 0 && (bytes[cut] & 0xC0) == 0x80 { cut -= 1 }
        let truncated = String(decoding: bytes.prefix(cut), as: UTF8.self)
        return (truncated, true)
    }

    static func contentHash(of s: String) -> String {
        let trimmed = s.trimmingCharacters(in: .whitespacesAndNewlines)
        let digest = SHA256.hash(data: Data(trimmed.utf8))
        return digest.map { String(format: "%02x", $0) }.joined()
    }
}
```

```swift
import Foundation
import GRDB

final class HistoryStore {
    let pool: DatabasePool

    init(path: String) throws {
        var config = Configuration()
        config.prepareDatabase { db in try db.execute(sql: "PRAGMA journal_mode=WAL") }
        self.pool = try DatabasePool(path: path, configuration: config)
        var migrator = DatabaseMigrator()
        Migrations.register(&migrator)
        try migrator.migrate(pool)
    }

    static func inMemory() throws -> HistoryStore {
        // For tests. Use temp file because DatabasePool requires file path.
        let tmp = NSTemporaryDirectory() + UUID().uuidString + ".sqlite"
        return try HistoryStore(path: tmp)
    }

    @discardableResult
    func insert(_ item: ClipItem) throws -> Int64 {
        try pool.write { db in
            try db.execute(sql: """
                INSERT INTO items (content, content_hash, source_bundle_id, source_app_name,
                                   created_at, pinned, byte_size, truncated)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?)
            """, arguments: [item.content, item.contentHash, item.sourceBundleID,
                             item.sourceAppName, item.createdAt, item.pinned ? 1 : 0,
                             item.byteSize, item.truncated ? 1 : 0])
            return db.lastInsertedRowID
        }
    }

    func listRecent(limit: Int = 50) throws -> [ClipItem] {
        try pool.read { db in
            try Row.fetchAll(db, sql: """
                SELECT * FROM items
                ORDER BY pinned DESC, created_at DESC
                LIMIT ?
            """, arguments: [limit]).map(Self.itemFromRow)
        }
    }

    static func itemFromRow(_ row: Row) -> ClipItem {
        ClipItem(
            id: row["id"], content: row["content"], contentHash: row["content_hash"],
            sourceBundleID: row["source_bundle_id"], sourceAppName: row["source_app_name"],
            createdAt: row["created_at"], pinned: (row["pinned"] as Int64) != 0,
            byteSize: row["byte_size"], truncated: (row["truncated"] as Int64) != 0
        )
    }
}
```

```swift
func prune(now: Int64, maxCount: Int, maxAgeSeconds: Int64) throws {
    try pool.write { db in
        try db.execute(sql: """
            DELETE FROM items
            WHERE pinned = 0
              AND id NOT IN (
                  SELECT id FROM items WHERE pinned = 0
                  ORDER BY created_at DESC LIMIT ?

            arguments: [now - maxAgeSeconds])
    }
}
```

```swift
import AppKit

struct PrivacyFilter {
    var respectConcealed = true
    var respectTransient = true
    var respectAutoGenerated = true

    static let concealedUTI = NSPasteboard.PasteboardType("org.nspasteboard.ConcealedType")
    static let transientUTI = NSPasteboard.PasteboardType("org.nspasteboard.TransientType")
    static let autoGenUTI   = NSPasteboard.PasteboardType("org.nspasteboard.AutoGeneratedType")
    static let internalUTI  = NSPasteboard.PasteboardType("com.zyw.clip.internal-paste")

    /// Return reason if filtered, nil if accepted.
    func reasonToSkip(types: Set<NSPasteboard.PasteboardType>, content: String,
                      sourceBundleID: String?, blacklist: Set<String>) -> String? {
        if types.contains(Self.internalUTI) { return "internal-paste" }
        if respectConcealed && types.contains(Self.concealedUTI) { return "concealed" }
        if respectTransient && types.contains(Self.transientUTI) { return "transient" }
        if respectAutoGenerated && types.contains(Self.autoGenUTI) { return "auto-generated" }
        if content.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty { return "empty" }
        if let bid = sourceBundleID, blacklist.contains(bid) { return "blacklisted" }
        return nil
    }
}
```

### Task 15: PasteboardObserver core

- [ ] Implementation (no timer yet; expose `tick(now:)` for tests):

```swift
import AppKit

final class PasteboardObserver {
    private let source: PasteboardSource
    private let store: HistoryStore
    private let filter: () -> PrivacyFilter
    private let blacklist: () -> Set<String>
    private let frontmost: () -> (bundleID: String?, name: String?)
    private(set) var lastChangeCount: Int

    static let maxBytes = 256 * 1024
    static let hardSkipBytes = 5 * 1024 * 1024

    init(source: PasteboardSource, store: HistoryStore,
         filter: @escaping () -> PrivacyFilter,
         blacklist: @escaping () -> Set<String>,
         frontmost: @escaping () -> (bundleID: String?, name: String?))
    {
        self.source = source; self.store = store
        self.filter = filter; self.blacklist = blacklist; self.frontmost = frontmost
        self.lastChangeCount = source.changeCount
    }

    /// Returns true if a new item was inserted/promoted. `now` injected for tests.
    @discardableResult
    func tick(now: Int64) throws -> Bool {
        let cc = source.changeCount
        guard cc != lastChangeCount else { return false }
        lastChangeCount = cc

        let types = Set(source.types())
        // Hard size guard before reading the string body.
        if let size = source.data(forType: .string)?.count, size > Self.hardSkipBytes {
            return false
        }
        guard let raw = source.string(forType: .string) else { return false }

        let app = frontmost()
        if let reason = filter().reasonToSkip(types: types, content: raw,
                                              sourceBundleID: app.bundleID,
                                              blacklist: blacklist()) {
            _ = reason  // hook for logging later
            return false
        }

        let (content, truncated) = ClipItem.truncateIfNeeded(raw, limit: Self.maxBytes)
```

**Files:** Modify `PasteboardObserver.swift`

```swift
private var timer: DispatchSourceTimer?
private var lastChangeAt: Int64 = 0
private var paused = false
private let queue = DispatchQueue(label: "clip.pasteboard.observer", qos: .utility)

func start() {
    let t = DispatchSource.makeTimerSource(queue: queue)
    t.setEventHandler { [weak self] in self?.fire() }
    rescheduleTimer(t: t, interval: 1.0)
    t.resume()
    self.timer = t
    NotificationCenter.default.addObserver(forName: NSWorkspace.willSleepNotification,
        object: nil, queue: nil) { [weak self] _ in self?.paused = true }
    NotificationCenter.default.addObserver(forName: NSWorkspace.didWakeNotification,
        object: nil, queue: nil) { [weak self] _ in
            self?.resetBaseline(); self?.paused = false
        }
    let dnc = DistributedNotificationCenter.default()
    dnc.addObserver(forName: .init("com.apple.screenIsLocked"), object: nil, queue: nil) {
        [weak self] _ in self?.paused = true
    }
    dnc.addObserver(forName: .init("com.apple.screenIsUnlocked"), object: nil, queue: nil) {
        [weak self] _ in self?.resetBaseline(); self?.paused = false
    }
}

private func fire() {
    guard !paused else { return }
    let now = Int64(Date().timeIntervalSince1970)
    do {
        let changed = try tick(now: now)
        if changed { lastChangeAt = now }
        if let t = timer { rescheduleTimer(t: t, interval: pickInterval(now: now)) }
    } catch {
        // log only
    }
}

private func pickInterval(now: Int64) -> TimeInterval {
    let ago = now - lastChangeAt
    if ago < 60 { return 0.5 }
    if ago > 300 { return 3.0 }
    return 1.0
}

private func rescheduleTimer(t: DispatchSourceTimer, interval: TimeInterval) {
    t.schedule(deadline: .now() + interval, repeating: interval, leeway: .milliseconds(100))
}
```

```swift
import SwiftUI
import AppKit

@main
struct ClipApp: App {
    @NSApplicationDelegateAdaptor(AppDelegate.self) private var delegate
    var body: some Scene {
        Settings { PreferencesWindow() }
    }

final class AppDelegate: NSObject, NSApplicationDelegate {
    var store: HistoryStore!
    var blacklist: BlacklistService!

    var hotkey: HotkeyManager!
    var statusItem: StatusItemController!
    var panelModel: PanelModel!

    var injector = PasteInjector()
    var pruner: DispatchSourceTimer!

    func applicationDidFinishLaunching(_ note: Notification) {
        NSApplication.shared.setActivationPolicy(.accessory)

        // Single instance.
        let mine = Bundle.main.bundleIdentifier ?? "com.zyw.clip"
        let dupes = NSRunningApplication.runningApplications(withBundleIdentifier: mine)
            .filter { $0.processIdentifier != ProcessInfo.processInfo.processIdentifier }
        if let other = dupes.first {
            other.activate()
            NSApp.terminate(nil)
            return
        }

        let dbPath = Self.databasePath()
        store = try! HistoryStore(path: dbPath)
        blacklist = BlacklistService(store: store)

            return PrivacyFilter(
                respectConcealed:    d.object(forKey: "clip.privacy.respectConcealed")    as? Bool ?? true,
                respectTransient:    d.object(forKey: "clip.privacy.respectTransient")    as? Bool ?? true,
                respectAutoGenerated:d.object(forKey: "clip.privacy.respectAutoGenerated") as? Bool ?? true)
        }
        let blacklistBuilder: () -> Set<String> = { [weak self] in
            (try? self?.blacklist.currentSet()) ?? []
        }
        observer = PasteboardObserver(
            source: NSPasteboardSource(),
            store: store,
            filter: filterBuilder, blacklist: blacklistBuilder,
            frontmost: { let a = NSWorkspace.shared.frontmostApplication
                         return (a?.bundleIdentifier, a?.localizedName) })
        observer.start()

        hotkey = HotkeyManager()
        hotkey.onToggle = { [weak self] in self?.togglePanel() }
        hotkey.install()

        panelModel = PanelModel(store: store)
        panel = PanelWindow()
        panel.setRoot(PanelView(model: panelModel))

        statusItem = StatusItemController()
        statusItem.onOpenPanel = { [weak self] in self?.togglePanel() }
        statusItem.onTogglePause = { [weak self] in self?.togglePause() }
        statusItem.onOpenPreferences = {
            NSApp.sendAction(Selector(("showSettingsWindow:")), to: nil, from: nil)
        }

        startPruner()
        showOnboardingIfNeeded()
    }

    private func togglePanel() {
        if panel.isVisible { panel.close() } else { panel.showAtCursor(); panelModel.reload() }
    }

    private func togglePause() {
        statusItem.isPaused.toggle()
        if statusItem.isPaused { observer.pause() } else { observer.resume() }
```
